.MODEL SMALL
.STACK 100H
.DATA
    Msg DB 'Hello! ', 'bollo ','ckoolo '

.CODE
MAIN PROC
    MOV AX, @DATA
    MOV DS, AX

    MOV si, OFFSET Msg  
PrintLoop:
    MOV AL, [si]        
    CMP AL, 0           
    JE Exit
    
    MOV AH, 2
    MOV DL, AL      
    INT 21H 
    
  
    
    INC si
    JMP PrintLoop 
    
 

Exit:
    MOV AH, 4CH
    INT 21H
MAIN ENDP
END MAIN
